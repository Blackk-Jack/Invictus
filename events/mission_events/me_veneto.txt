	        add_opinion = {

	            target = root
	        }
	    }
	    scope:country.root = {
	        add_opinion = {
	            modifier = venetic_cenomani_alliance_opinion
	            target = c:CNM
	        }

	        if = {
	            limit = {
	                alliance_with = c:HIS
	            }        
	        }
	        trigger_event = { id = secure_venetia_mission_event.16 } #Option to break alliance with The Histri
	    }
	}
}
