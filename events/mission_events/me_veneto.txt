secure_venetia_mission_event.22 = { #Rome wishes further integration (from perspective of veneto) 
    type = country_event
    title = secure_venetia_mission_event.22.t
    desc = secure_venetia_mission_event.22.desc
    picture = romans_marching

    option = { #Very well, let us integrate further with rome!
        name = secure_venetia_mission_event.22.a 
        custom_tooltip = secure_venetia_mission_event.22.a.tt
        hidden_effect = {
        change_country_color = roman_red
        change_government = aristocratic_republic
        root = {
            if = {   
                limit = {
    	            area:histria_area = {
    		            any_area_province = {
	    		            count = all
	    		            owned_or_subject_owned = root
    		            }
    	            }
                }
                change_country_name = VENETIA_ET_HISTRIA
            }
            else = {
                change_country_name = VENETIA
            }
        }
        if = {
            limit = {
                NOT = {
                    has_variable = veneto_adopt_roman_culture
                }
            }
            trigger_event = { id = secure_venetia_mission_event.21 days = { 30 3600 }}
        }
        c:ROM = {
            release_subject = c:VNT
            make_subject = {
                target = c:VNT
                type = feudatory
            }
        }
       }
    }
    option = { #This demand is a bit much, even from Rome
        name = secure_venetia_mission_event.22.b
        custom_tooltip = secure_venetia_mission_event.22.b.tt 
        add_stability = -20
        add_centralization = -20
        hidden_effect = {
         c:ROM = {
            trigger_event = {
                id = secure_venetia_mission_event.23            
            }
        }       
        }
    }
}
