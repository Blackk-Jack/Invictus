# Recursive "control" event for Tibet
me_nabhapakti.13 = {
    type = country_event
    hidden = yes
    immediate = {
        change_variable = {
            name = tibetan_vassal_var
            add = 1
        }
        random_list = {
            3 = {
                trigger = {
                    var:tibetan_vassal_var <= 3
                    NOT = { always = no } # This is the "fail" trigger
                }
                trigger_event = me_nabhapakti.130
            }
            3 = {
                trigger = {
                    var:tibetan_vassal_var <= 3
                    NOT = { always = no } # This is the "fail" trigger
                }
                trigger_event = me_nabhapakti.131
            }
            3 = {
                trigger = {
                    var:tibetan_vassal_var <= 3
                    NOT = { always = no } # This is the "fail" trigger
                }
                trigger_event = me_nabhapakti.132
            }
            1 = {
                trigger = {
                    always = no # This is the "fail" trigger
                }
                # Fail effects...
                remove_variable = is_doing_nepalese_task
            }
            1 = {
                trigger = {
                    NOT = { always = no } # This is the "fail" trigger
                }
                # final event/effect that concludes that chain is here. This is basically the "timeout"
                remove_variable = is_doing_nepalese_task
                trigger_event = me_nabhapakti.134
                add_country_modifier = {
                    name = nepalese_vassal_mod
                    duration = 730
                    mode = add_and_extend
                }
            }
        }
    }
}
