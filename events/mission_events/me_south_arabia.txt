    }

        set_variable = accepted_israelite_communities
        if = {
            limit = {
                var:israelite_communities_events_var >= 4 # out of 5 max
            }
            add_innovation = 2
            while = {
                count = 15
                random_owned_province = {
                    random_pops_in_province = {
                        limit = {
                            NOT = { pop_religion = judaism }
                        }
                        set_pop_religion = judaism
                    }
                }
            }
        }
        else_if = {
            limit = {
                var:israelite_communities_events_var >= 2 # effective end is 3
            }
            add_innovation = 2
            while = {
                count = 10
                random_owned_province = {
                    random_pops_in_province = {
                        limit = {
                            NOT = { pop_religion = judaism }
                        }
                        set_pop_religion = judaism
                    }
                }
            }
        }
        else_if = {
            limit = {
                var:israelite_communities_events_var >= 0 # effective end is 1
            }
            add_innovation = 1
            while = {
                count = 5
                random_owned_province = {
                    random_pops_in_province = {
                        limit = {
                            NOT = { pop_religion = judaism }
                        }
                        set_pop_religion = judaism
                    }
                }
            }
        }
        else_if = {
            limit = {
                var:israelite_communities_events_var < 0 
            }
            add_political_influence = -75
        }
        else = {
            add_political_influence = -100 # fallback, should not actually ever happen
        }
    }

    option = { # tolerate
    	name = south_arabia.40.b
        if = {
            limit = { var:israelite_communities_events_var >= 0 }
            add_stability = 10
            add_country_modifier = {
                name = accepted_israelite_communities_cmod
                duration = 9125 # 25 years
            }
        }
        else_if = {
            limit = { var:israelite_communities_events_var >= -2 }
            add_stability = 5
            add_country_modifier = {
                name = accepted_israelite_communities_cmod
                duration = 3650 # 10 years
            }
        }
        else = {
            add_political_influence = -75
            add_country_modifier = {
                name = accepted_israelite_communities_cmod
                duration = 1825 # 5 years
            }
        }
    }

    option = { # expel
    	name = south_arabia.40.c
        custom_tooltip = expelled_pops_move_to_judea_tt
        hidden_effect = {
            every_owned_province = {
                every_pops_in_province = {
                    limit = {
                        pop_religion = judaism
                    }
                    move_pop = p:687 # Jerusalem
                }
            }
        }
        if = {
            limit = {
                p:687.owner = {
                    religion = judaism
                }
            }
            p:687.owner = {
                add_opinion = {
                    modifier = expelled_israelites
                    target = root
                }
            }
        }
        add_treasury = 400
        if = {
            limit = { var:israelite_communities_events_var <= -3 }
            add_military_experience = 50
            add_political_influence = 50
        }
        else_if = {
            limit = { var:israelite_communities_events_var < -1 }
            add_military_experience = 25
            add_political_influence = 25
        }
        else_if = {
            limit = { var:israelite_communities_events_var >= 0 }
            add_political_influence = -100
            add_military_experience = 15
        }
    }
}
